 (defn greatest-product-five-digits [big-digits]
   "Find the greatest product of five consecutive digits in the
   1000-digit number"
 
   (->> big-digits
        (filter #(Character/isDigit %))      ;; Clean up \n character
        (map #(Character/getNumericValue %)) ;; From character to
                                             ;; number, i.e. from \8 to
                                             ;; 8
        (partition 5 1)                      ;; Partition by 5 per
                                             ;; group, step by 1
        (map (partial reduce *))             ;; Multiply each group
        (reduce max)))                       ;; Find the greatest product.
